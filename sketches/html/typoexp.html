function mouseMoved() {
    if (lastX === 0 && lastY === 0) {
        lastX = mouseX;
        lastY = mouseY;
    }

    let word = words[wordIndex];
    let d = dist(mouseX, mouseY,
                 lastX, lastY);

    // Draw a word only when the mouse
    // has moved far enough
    if (d > textWidth(word) + 20) {
        let angle = atan2(
            mouseY - lastY,
            mouseX - lastX
        );

        push();
        translate(mouseX, mouseY);
        rotate(angle);

        const colors = ['#E9F1F7',
          '#2274A5', '#E7DFC6', '#816C61'];
        fill(colors[wordIndex
             % colors.length]);
        noStroke();

        text(word, 0, 0);
        pop();

        lastX = mouseX;
        lastY = mouseY;
        wordIndex = (wordIndex + 1)
                    % words.length;
    }
}

La phrase est une longue suite de mots découpée avec split(" "). On garde la position du dernier mouvement de souris (lastX, lastY).

On regarde la distance entre l'ancienne et la nouvelle position. Si le mouvement est plus grand que la largeur du mot + 20, on dessine un mot — cela espace naturellement le texte.

let angle = atan2(mouseY - lastY, mouseX - lastX); calcule l'angle du mouvement. Ensuite rotate(angle) adapte l'orientation du mot selon la direction du pointeur.

Les couleurs alternent dans la palette grâce à wordIndex % colors.length, créant un ruban typographique multicolore.
