Moire par erreur

function setup() {
    frameRate(3);
    createCanvas(400, 400);
    background('#131B23');
}

var paul = 120;

function draw() {
    noFill();

    let strokeCol = palette[
      Math.floor(paul / 5) % palette.length
    ];
    stroke(strokeCol);
    strokeWeight(1.5);

    circle(200, 60, paul);
    circle(200, 60, paul);
    circle(200, 60, paul);
    circle(100, 60, paul);

    paul = paul + 5;
}
My Canva
Paul est 120 — c'est la taille de départ du cercle. On dit le centre "200, 60". La fonction draw() est une boucle qui recommence sans cesse. Paul est la taille, et on fait : paul = paul + 5; — donc on agrandit le cercle à chaque tour. On peut choisir le centre puis avoir plusieurs variations. On dessine 3 fois le même cercle au même endroit : on voit surtout un seul cercle, mais il est redessiné plusieurs fois, ce qui crée un effet moiré inattendu.
