Bouncing Particles

function setup() {
    createCanvas(400, 400);
    for (let i = 0; i < numParticles; i++) {
        particles.push({
            x: random(width),
            y: random(height),
            vx: random(-2.5, 2.5),
            vy: random(-2.5, 2.5),
            size: random(8, 22),
            color: color(random(palette))
        });
    }
}

function mousePressed() {
    for (let p of particles) {
        let angle = atan2(
            mouseY - p.y, mouseX - p.x
        );
        let pullForce = 3.5;
        p.vx += cos(angle) * pullForce;
        p.vy += sin(angle) * pullForce;
    }
}
My Canva
90 particules colorées rebondissent dans le canvas. Un fond semi-transparent crée des traînées de mouvement fluides. En cliquant, toutes les particules sont attirées vers la position du clic avec une force directionnelle calculée avec atan2 — une fonction qui trouve l'angle entre deux points.
